## 支付结果回调监听

```lua
import "org.json.JSONObject"

local SERVER_URL = "https://your-server.com/notify"

function sendNotify(json)
    local urlCls = findClass("java.net.URL", hostLoader)
    local url = newInstance(urlCls, SERVER_URL)
    local conn = callMethod(url, "openConnection")
    callMethod(conn, "setRequestMethod", "POST")
    callMethod(conn, "setDoOutput", true)
    callMethod(conn, "setRequestProperty", "Content-Type", "application/json")
    local out = callMethod(conn, "getOutputStream")
    callMethod(out, "write", callMethod(json, "toString"):getBytes("UTF-8"))
    callMethod(out, "close")
    log("服务端响应码:", callMethod(conn, "getResponseCode"))
end

local payClazz = findClass()
    :pkg("com.example.pay")
    :usingStrings("payResult", "tradeNo")
    :singleOrNull()

if payClazz == nil then
    log("未找到支付类，请更新匹配条件")
    return
end

hookMethodBefore(payClazz, "onPayResult", function(param)
    local intent = callMethod(param.thisObject, "getIntent")
    local extras = callMethod(intent, "getExtras")
    local result = callMethod(extras, "getString", "pay_result")
    if result ~= nil then
        log("支付结果:", result)
        local json = newInstance(JSONObject)
        callMethod(json, "put", "result", result)
        callMethod(json, "put", "appVersion", hostVerName)
        sendNotify(json)
    end
end)

toast("支付监听已启动")
```

import "com.iapp.app.HomeApp"

hookMethodBefore(HomeApp, "p0", function(param)

    setStaticObjectField(c, "r0", {})

## 拦截并修改返回值

### replace 模式

```lua
hookReplace(UserManager, "isVIP", function(param)
    param.result = true
end)
```

### after 模式

```lua
hookMethodAfter(UserManager, "getLevel", function(param)
    if param.result ~= nil and param.result < 10 then
        log("等级提升:", param.result, "->", 10)
        param.result = 10
    end
end)
```

## 附加字段跟踪计数

```lua
hookMethodBefore(TargetClass, "onClick", function(param)
    local obj = param.thisObject
    local count = getAdditionalField(obj, "clickCount") or 0
    count = count + 1
    setAdditionalField(obj, "clickCount", count)
    log("点击次数:", count)
end)
```

## 性能与最佳实践

### Hook 注意事项

1. 只 Hook 必要的方法，每个 Hook 有性能开销。
2. Hook 回调在宿主主线程执行，不要做耗时操作。
3. **避免死循环**：Hook 回调中不要调用被 Hook 的同一个方法。

```lua
hookMethodBefore(MyClass, "setData", function(param)
    param:setArg(0, "modified")
end)
```

4. 用完及时 `unhook(id)` 清理。

### 反射优化

- 在脚本顶层缓存 Class 对象，不要在回调中反复 `findClass`。
- `findField` / `findMethodExact` 内部已有缓存，重复调用安全。
- 优先使用类型化方法（`getIntField` 比 `getObjectField` + 强转更快）。

### 常见陷阱

| 陷阱 | 说明 | 对策 |
|------|------|------|
| 混淆 | 目标应用类名每版可能变化 | 使用 DexKit，不硬编码类名 |
| 多进程 | 目标应用可能有多个进程 | 在项目设置中选择正确进程模式 |
| 语法错误 | Lua/JS/Bsh 语法差异 | 确认入口文件扩展名正确（main.java / main.lua / main.js） |

### 三语言语法对照

| 含义 | Bsh (Java) | Lua | Js |
|------|------------|-----|-----|
| 字符串拼接 | `"a" + "b"` | `"a" .. "b"` | `"a" + "b"` |
| 相等 | `==` | `==` | `===` |
| 不等 | `!=` | `~=` | `!==` |
| 与 | `&&` | `and` | `&&` |
| 或 | `\|\|` | `or` | `\|\|` |
| 非 | `!` | `not` | `!` |
| 注释 | `//` | `--` | `//` |
| 空值 | `null` | `nil` | `null` |
| 数组索引起始 | `arr[0]` | `arr[1]` | `arr[0]` |
| 函数定义 | `void fn(){}` | `function fn() end` | `function fn(){}` |
| 字符串长度 | `str.length()` | `#str` | `str.length` |

| 函数 | 说明 |
|------|------|
| `hookMethodBefore(clazz, name, [id], cb)` | 调用前 Hook |
